from sys.database_files
where type_desc='ROWS';

create partition function [MonthlyPartition] (datetime)
as range right for values('20180201','20180301','20180401','20180501','20180601','20180701','20180801','20180901','20181001','20181101','20181201');

create partition scheme MonthWisePartition
as partition MonthlyPartition to (January,February,March,April,May,June,July,August,September,October,November,December);

create table sqlinsert
 (empid int identity(1,1) not null,
  firstname nvarchar(255) null,
  lastname nvarchar(255) null,
  occupation nvarchar(255) null,
  yearlyincome float null,
  sales float null,
  insertdate datetime null
  )on MonthWisePartition(insertdate);
