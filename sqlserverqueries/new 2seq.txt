CREATE SEQUENCE receipt_no
START WITH 1
INCREMENT BY 1;

create table goods_receipts
(
 receipt_id int primary key default(next value for receipt_no),
 order_id int not null,
  full_receipt bit not null,
    receipt_date date not null,
    note nvarchar(100),
);

create table invoice_receipts
(
 receipt_id int primary key
        default (next value for receipt_no), 
    order_id int not null, 
    is_late bit not null,
    receipt_date date not null,
    note nvarchar(100)
);
